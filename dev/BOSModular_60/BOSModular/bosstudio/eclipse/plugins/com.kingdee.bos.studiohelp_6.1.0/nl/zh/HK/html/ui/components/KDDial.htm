使用指南

表盤控件的程序包括一個靈活的基類和KPI表盤實例。基類的使用可以參見示例。下面主要是KPI表盤的使用。

創建一個KPI表盤可以使用KDDial類。它的構造方法有以下參數：

·range: 總範圍

·normal: 正常範圍

·warning: 警告範圍

·critical: 危急範圍

·title: 標題

如果只需要一個總範圍而不需要三個範圍的話，可以使用以下構造函數:

KDDial(KpiRange range, String title);

下面是表盤的方法說明：

1．  對表盤的值進行設置。使用方法：

              KDDail.setValue(Object value);

              value: KDDail中取Double類型。

2．  對表盤外觀的定制

取得表盤中的組件對象，進行屬性設置。KPI表盤中的組件默認序號為0。

例子1：設置刻度盤不同範圍的顏色

a. 獲得刻度盤對像：KpiScale scl = KDDial.getScale();

b. 定制某個範圍的顏色：scl.setNormalPaint(Color.Red);

例子2：設置刻度盤標籤顯示格式

dial.getScale().setTickLabelFormat(new DecimalFormat("#"));

默認顯示格式為：DecimalFormat("###』%』");

3．  定制多個範圍

如果需要對刻度盤添加多個範圍使用KpiScale.addRange(KpiRange range, int type)方法，其中type參數是區域類型，包含以下類型：

· 總範圍類型 TOTAL

· 正常範圍類型 NORMAL

· 警戒範圍類型 WARNING

· 危急範圍類型 CRITICAL

如果某個範圍中包含無窮大/小的情況，使用KpiRange.INFINITY或KpiRange.INFINITESIMAL構建KpiRange，然後再添加到Scale中。修改某一類型區域的範圍可以通過以下方法進行：

1．  調用clearRanges()清除某一類型區域所有的範圍;

2．  調用addRange()重新設置該區域要包含的範圍.

獲得某個類型的所有範圍使用KpiScale.getRanges()。刪除某個特定範圍使用方法KpiScale.removeRange()。

添加範圍的例子如下（其中min,max是兩個JSpinner, comboRanges是一個JComboBox）：

    public void addRange() {

       RangeItem item = (RangeItem) comboRanges.getSelectedItem();

       KpiScale scl = dial.getScale();

       KpiRange range = null;

       double minv = ((Number) min.getValue()).doubleValue();

       double maxv = ((Number) max.getValue()).doubleValue();

       range = new KpiRange(minv, maxv);

       scl.addRange(range, item.getType());

       updateIndex();

       dial.repaint();

}

4.設置表盤單位標題:

      使用接口:public void setUnitTitle(String unitTtl);

      例:

      String ttl = 「(單位:萬元)」;

      dial.setUnitTitle(ttl);

示例

示例1：KPI表盤(KDDial)的使用

public class TestKDDial extends TestingFrame {

    public TestKDDial() {

       KpiRange total = new KpiRange(0, 120);

       KpiRange normal = new KpiRange(0, 30);

       KpiRange warning = new KpiRange(30, 85);

       KpiRange critical = new KpiRange(85, 120);

       KDDial dial =

           new KDDial(

              total,

              normal,

              warning,

              critical,

              "利潤完成率");

        Dial.setUnitTitle("（單位：萬元）");//設置單位標題

       dial.setValue(new Double(60));

       cp.add(dial);

    }

   

    public static void main(String[] args) {

       TestKDDial test = new TestKDDial();

       test.show();

    }

}

 

示例2: 基礎表盤的創建和配置

public class TestBasicDial extends TestingFrame {

 

    BasicDial dial = null;

 

    public TestBasicDial() {

       // 軸心

       Point[] pts =

           {

              new Point(100, 100),

              new Point(150, 80),

              new Point(120, 150),

              new Point(70, 80)};

      

       // 半徑

       int[] r = { 80, 30, 40, 15 };

      

       // 指針s

       Object[][] ndls = { { "needle0", new DialNeedle(pts[0], r[0])}, {

              "needle1", new DialNeedle(pts[1], r[1])

              }, {

              "needle2", new DialNeedle(pts[2], r[2])

              }

       };

 

       // 刻度盤

       Object[][] scls =

           {

              {

                  "scale0",

                  new DialScale(

                     pts[0],

                     r[0],

                     -40,

                     160,

                     new Double(-40),

                     new Double(120))},

              {

              "scale1",

                  new DialScale(

                     pts[1],

                     r[1],

                     0,

                     360,

                     new Double(0),

                     new Double(360))

              },

                  {

              "scale2",

                  new DialScale(

                     pts[2],

                     r[2],

                     240,

                     240,

                     new Double(240),

                     new Double(480),

                     false)

              },

                  {

              "scale3",

                  new DialScale(

                     pts[3],

                     r[3],

                     -45,

                     300,

                     new Double(-45),

                     new Double(255))

               }

       };

 

       // 刻度盤的標題

       DialScale scl = null;

       for (int i = 0; i < scls.length; i++) {

           scl = (DialScale) scls[i][1];

           scl.setCaption("D" + i);

       }

 

       // 指針刻度盤對應關係

       Object[][] _n2s = { { "needle0", "scale0" }, {

              "needle1", "scale1" }, {

              "needle2", "scale2" }

       };

       Map n2s = new HashMap();

              for (int i = 0; i < _n2s.length; i++) {

                  n2s.put(_n2s[i][0], _n2s[i][1]);

              }

             

       // 顯示器

       Object[][] inds = { { "indicator0", new DialIndicator()}

       };

 

       ((DialIndicator) inds[0][1]).setBounds(5, 15, 150, 19);

 

       // 底盤

       DialPan pan = new DialPan();

      

       // 標題

       String title = "Dial Title";

 

       // 創建一個多指針多刻度盤的表盤

       dial = DialFactory.createComplexDial(ndls, scls, n2s, inds, pan, title);

      

       // 對每個指針賦值

       dial.setValue("needle0", new Integer(30));

       dial.setValue("needle1", new Integer(127));

       dial.setValue("needle2", new Integer(300));

 

       cp.add(dial);

       //this.setContentPane(new MyPanel());

 

       JButton btn = new JButton("button");

       btn.addActionListener(new ButtonAction());

       cp.add(btn);

    }

 

    public class MyPanel extends JPanel {
