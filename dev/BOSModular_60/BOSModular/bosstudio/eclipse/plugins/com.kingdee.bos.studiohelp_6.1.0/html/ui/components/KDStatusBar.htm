功能列表

  状态栏可用了！

 

 

固定在窗口的底部，有合适高度，宽度与所在窗口的宽度相同

不可移动，不可以浮动窗口方式显示。

按照Word和永中office的状态栏风格展现状态栏上的各区域。

1，  区域 指的是状态栏上的容器，能够容纳一个或多个状态控件（ 见图二中的红圈 ）。区域的外观默认情况下和word风格一样，为凹下去的，但可通过get得到区域，进行设置Border风格。

2，  状态控件 见图二中的蓝圈。状态控件的外观默认情况下和word一样，为平铺的。

3，  区域大小根据容纳的状态控件的首先大小进行设定。

4，  状态控件大小在构造时指定首选大小。状态控件大小有默认大小设置。

提供根据key值，访问区域的方法。区域的key值是唯一的。

区域的布局：

1，  默认情况  靠左布局（类似JTextField中的文字可以靠左，居中，靠右），从左至右布局，采用先添加，先靠左添加依次布局。推荐使用默认布局，用户不需去设置区域的布局，区域很自然的就能布局到word效果。

2，  区域也可 靠右布局  ，但从左至右布局，采用先添加，先靠左添加依次布局。

3，  可设置区域的间距。

状态控件是JComponent类型； 默认情况下为JLabel；可设置图标，可设置文字；状态控件可定制，但需实现状态控件的接口，以能实现文字或图标的可设。

提供根据key值，访问状态控件的方法。Key值是唯一的。

状态控件可set/getUserObject()

状态控件可set/get 激活的图标和钝化的图标。

状态控件可 set/get 字符串标题。

状态控件可给监听ActionListener事件。

状态控件可设置是否可见

状态控件上方可缓慢（可供开发者设置delay）的弹出一个小面板。该小面板可被开发者添加任何Swing控件。

 

状态控件可设置激活和钝化

区域的Border可以设置，默认情况为凹下去的。

    /**

     *  构造状态栏

     * 目前状态栏已经支持了

     **/

    private KDStatusBar getStatusBar() {

       if (statusBar == null) {

           //1 创建状态栏

           statusBar = new KDStatusBar();

          // 2 创建状态栏上的一个区域

           SimpleStatusBarPart part1 = new SimpleStatusBarPart(statusBar);

           part1.setPreferredSize(new Dimension(200,30));

          //2.1 设置该区域的列间距

           part1.setHgrap(0);

           //2.2 创建一个renderer，该renderer是被添加到状态栏里的区域里

           SimpleStatusBarRenderer render10 = new SimpleStatusBarRenderer();

           //2.3 设置renderer的文本信息

           render10.setText("111");

           //2.4 配置该renderer里的内容。

           configPopup(render10.getPopupPane(),0);

          

          // 3 创建一个renderer，该renderer是被添加到状态栏里的区域里

           SimpleStatusBarRenderer render11 = new SimpleStatusBarRenderer();

           // 。。。。

           render11.setText("222");

           configPopup(render11.getPopupPane(),1);

          

           SimpleStatusBarPart part2 = new SimpleStatusBarPart(statusBar);

           part2.setPreferredSize(new Dimension(200,30));

           SimpleStatusBarRenderer render20 = new SimpleStatusBarRenderer();

           SimpleStatusBarRenderer render21 = new SimpleStatusBarRenderer();

           render20.setText("333");

           configPopup(render20.getPopupPane(),2);

           render21.setText("444");

           configPopup(render21.getPopupPane(),3);

          

           SimpleStatusBarPart part3 = new SimpleStatusBarPart(statusBar);

           part3.setPreferredSize(new Dimension(150,30));

           SimpleStatusBarRenderer render30 = new SimpleStatusBarRenderer();

           part3.addRenderer(render30,"r4");

           render30.setText("555");

           configPopup(render30.getPopupPane(),4);

          

           SimpleStatusBarPart part4 = new SimpleStatusBarPart(statusBar);

           part4.setPreferredSize(new Dimension(200,30));

           final SimpleStatusBarRenderer render40 = new SimpleStatusBarRenderer();

           part4.addRenderer(render40,"r5");

           render40.setText("666");

           configPopup(render40.getPopupPane(),5);

           //  监听renderer上鼠标事件，并弹出/隐藏renderer上的小面板

           render40.addMouseListener(new MouseAdapter(){

              public void mouseClicked(MouseEvent e){

                  if(render40.isPopupVisible()){

                     render40.setPopupVisible(false);

                  }else{

                     render40.setPopupVisible(true);

                  }

                                            

              }

           });

          

           //4 添加一个区域到该状态栏上去。

           statusBar.addPart(part1);         

           statusBar.addPart(part2);

           statusBar.addPart(part3);

           statusBar.addPart(part4);

 

          // 5 添加一个renderer到 区域上去。

           statusBar.addRendererToPart(render10,"r0",0);

           statusBar.addRendererToPart(render11,"r1",0);

           statusBar.addRendererToPart(render20,"r2",1);

           statusBar.addRendererToPart(render21,"r3",1);

          

          

          

       }

       return statusBar;

    }

   

 

    配置状态栏上rendeer的小面板的内容。

    private void configPopup(JPanel p,int index){

       p.setPreferredSize(new Dimension(100,300));

       p.setLayout(new BorderLayout());

      

       KDTextArea kta = new KDTextArea();

       kta.setLineWrap(true);

       kta.setText("render"+index+" \n您有邮件需要处理！\n 您的工作流上有四个审核单需要处理! \n 你明天需要去上海开会！ \n 你的手机需要缴费了！！！");

          

       kta.setEditable(false);

       p.add(kta,java.awt.BorderLayout.CENTER);

             

   }

 

 

 

    设置状态栏中指定 renderer的 文本。  

    private void settingStatusBarText(){

       String r = (String)getListRenderer().getSelectedValue();

       ISimpleStatusBarRenderer renderer =  getStatusBar().getRenderer(r);

       if(renderer!=null){

           renderer.setText(getRenderTextField().getText());

       }

    }

 

 

    弹出/隐藏 状态栏中某个renderer上的小面板。

    private void showHideStatusPopup(){

       String r = (String)getListRenderer().getSelectedValue();

       if(r==null || r.length()<=0) return;

       ISimpleStatusBarRenderer renderer =  getStatusBar().getRenderer(r);      

       if(renderer.isPopupVisible()){

           renderer.setPopupVisible(false);

       }else{

           renderer.setPopupVisible(true);          

       }

 

   }
