使用指南

 

//添加一个带标识符的工具栏，该工具栏默认布局在北区。

getMainFrame().addToolBar(getToolBarThird(), “TOOLBAR_THIRD”);

 

//删除窗体上所有的工具栏。

getMainFrame().removeAllToolBar();

 

// 添加一个工具栏到东区。

getMainFrame().addToolBar(getToolBarFirst(),KDToolBar.EAST,”TOOLBAR_FIRST”);

 

//删除一个标识符为“TOOLBAR_FIRST “的工具栏。

getMainFrame().removeToolBarByIdentify(“TOOLBAR_FIRST”);
